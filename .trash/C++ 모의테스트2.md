```c++
#include <fstream>
#include <iostream>
#include <vector>

int main(){

    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");

    int n,x;
    fin >> n;
    std::vector<int> arr(n);

    for(int& v : arr){
        fin >> v;
    }
    for(int v: arr){
        std::cout << v << " ";
    }

    fin >> x;
    std::cout << "\n" << x;


    int count = 0;

    for(int i=0; i<n; i++){
        for(int j=i+1; j<n; j++){
            if(arr[i]+arr[j]==x) count++;
        }
    }

    fout << count;

    return 0;
}
```

#include <iostream>
#include <fstream>
#include <unordered_set>
#include <vector>

int main(){
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");

    int n;
    fin >> n;
    std::cout << n << "\n";

    std::unordered_set<int> set;
    std::vector<int> arr;
    int num;

    for(int i=0; i<n; i++){
        fin >> num;
        std::cout << num << " ";
        if(set.insert(num).second){
            arr.push_back(num);
        }
    }

    for(int x:arr){
        fout << x << " ";
    }

    return 0;
}

# 3
```c++
#include <iostream>
#include <fstream>

int main(){
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");

    int n;
    fin >> n;

    int num;
    int positive=0,negative=0,zero=0;

    for(int i=0; i<n; i++){
        fin >> num;
        std::cout << num << " ";
        if(num==0) zero++;
        if(num>0) positive++;
        if(num<0) negative++;
    }

    fout << "Positive: " << positive << "\n";
    fout << "Negative: " << negative << "\n";
    fout << "Zero: " << zero;
  
    return 0;
}
```

#include <string>
#include <algorithm>

    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");

    std::reverse(str.begin(), str.end());
