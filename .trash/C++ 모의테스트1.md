```c++
#include <iostream>
#include <fstream>

int main(){
    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");

    int N;
    fin >> N;

    // 위쪽 영역 (1~N줄)
    for (int i = 1; i <= N; i++) {
        for (int j = 0; j < N - i + 1; j++) fout << " ";
        for (int k = 0; k < 2 * i - 1; k++) fout << "*";
        fout << "\n";
    }

    // 가운데 줄
    for (int k = 0; k < 2 * N + 1; k++) fout << "*";
    fout << "\n";

    // 아래쪽 영역 (N~1줄)
    for (int i = N; i >= 1; i--) {
        for (int j = 0; j < N - i + 1; j++) fout << " ";
        for (int k = 0; k < 2 * i - 1; k++) fout << "*";
        fout << "\n";
    }
    return 0;
}
```

```c++
#include <iostream>
#include <fstream>
#include <vector>

int main(){

    std::ifstream fin("input.txt");
    std::ofstream fout("output.txt");

    int n,x;
    fin >> n;
    std::vector<int> arr(n);
    for(int& num : arr){
        fin >> num;
    }
    fin >> x;

    int count = 0;
    for(int num : arr){
        if(num==x) count++;
    }
    fout << count;

    return 0;
}
```
